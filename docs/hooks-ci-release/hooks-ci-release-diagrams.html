2. Pre-Commit Hook

Runs on every git commit. Detects staged file categories and runs only relevant checks. Skips entirely if no files are staged. 

flowchart TD
  START(["git commit"]) --> DETECT["Detect staged files
git diff --cached --name-only"]

  DETECT --> EMPTY{"Any staged
files?"}
  EMPTY -- "NO" --> SKIP(["Exit 0
Skip all checks"])

  EMPTY -- "YES" --> CATEGORIZE["Categorize staged files"]

  CATEGORIZE --> CAT_CODE["HAS_CODE?
.ts, .tsx, .js, .jsx"]
  CATEGORIZE --> CAT_CSS["HAS_CSS?
.css"]
  CATEGORIZE --> CAT_WS["HAS_WORKSPACE?
package.json, pnpm-*,
.npmrc"]
  CATEGORIZE --> CAT_TEXT["HAS_TEXT?
.ts, .tsx, .js, .jsx,
.json, .md, .css,
.yaml, .yml"]

  CAT_CODE --> LINT_STAGED
  CAT_CSS --> LINT_STAGED
  CAT_WS --> LINT_STAGED
  CAT_TEXT --> LINT_STAGED

  subgraph LINT_STAGED["ALWAYS: lint-staged (staged files only)"]
    direction LR
    LS1["prettier --write
code + json + md + css"]
    LS2["secretlint
code files only"]
  end

  LINT_STAGED --> WS_CHECK{"HAS_WORKSPACE?"}
  WS_CHECK -- "YES" --> WS_CHECKS
  WS_CHECK -- "NO" --> CODE_CHECK

  subgraph WS_CHECKS["Workspace Checks"]

  end

  WS_CHECKS --> CODE_CHECK

  CODE_CHECK{"HAS_CODE?"}
  CODE_CHECK -- "YES" --> CODE_CHECKS
  CODE_CHECK -- "NO" --> CSS_CHECK

  subgraph CODE_CHECKS["Code Quality Checks"]

3. Pre-Push Hook

Runs on every git push. Detects change categories, validates Docker availability, and runs checks sequentially to avoid resource contention. 

flowchart TD
  START(["git push"]) --> BASE["Detect base branch
origin/develop or @{upstream}"]
  BASE --> DIFF["git diff --name-only
base..HEAD"]
  DIFF --> CAT["Categorize changes"]

  CAT --> DEPLOY_PAT["Deploy: Dockerfile,
docker/*, .dockerignore"]
  CAT --> CODE_PAT["Code: *.ts, *.tsx,
*.js, *.jsx"]

  DEPLOY_PAT --> CHECK_BOTH{"Deploy changes?
Code changes?"}

  CHECK_BOTH -- "Neither" --> SKIP(["No relevant changes
Skip all, exit 0"])

  CHECK_BOTH -- "Deploy only" --> DOCKER_PATH_1["Docker PATH discovery
Check 3 Windows paths"]
  DOCKER_PATH_1 --> DOCKER_INFO_1{"docker info
running?"}
  DOCKER_INFO_1 -- "NO" --> DOCKER_FAIL_1(["EXIT 1
Docker not running"])
  DOCKER_INFO_1 -- "YES" --> DOCKER_FG["Docker health check
FOREGROUND
bash scripts/docker-health-check.sh"]
  DOCKER_FG --> DOCKER_FG_EXIT(["Exit with Docker result"])

  CHECK_BOTH -- "Both deploy + code" --> DOCKER_PATH_2["Docker PATH discovery
Check 3 Windows paths"]
  DOCKER_PATH_2 --> DOCKER_INFO_2{"docker info
running?"}
  DOCKER_INFO_2 -- "NO" --> DOCKER_FAIL_2(["EXIT 1
Docker not running"])
  DOCKER_INFO_2 -- "YES" --> CODE_BOTH["Step 1: Code checks FIRST"]

  subgraph CODE_BOTH_BOX["Code Checks (no build - Docker covers it)"]
    direction TB
    CB1["pnpm typecheck"]
    CB2["pnpm test:coverage"]
    CB3["pnpm test:e2e"]
    CB1 --> CB2 --> CB3
  end
  CODE_BOTH --> CODE_BOTH_BOX

  CODE_BOTH_BOX --> CODE_BOTH_RESULT{"Code checks
passed?"}
  CODE_BOTH_RESULT -- "NO" --> CODE_FAIL(["EXIT 1"])

  CODE_BOTH_RESULT -- "YES" --> DOCKER_SEQ["Step 2: Docker health check
AFTER code checks
bash scripts/docker-health-check.sh"]
  DOCKER_SEQ --> DOCKER_RESULT{"Docker check
passed?"}
  DOCKER_RESULT -- "NO" --> DOCKER_BG_FAIL(["EXIT 1
Docker health failed"])
  DOCKER_RESULT -- "YES" --> BOTH_PASS(["All passed"])

  CHECK_BOTH -- "Code only" --> CODE_ONLY["Code checks + build"]
  subgraph CODE_ONLY_BOX["Code Checks (includes build)"]
    direction TB
    CO1["pnpm typecheck"]
    CO2["pnpm test:coverage"]
    CO3["pnpm build"]
    CO4["pnpm test:e2e"]
    CO1 --> CO2 --> CO3 --> CO4
  end
  CODE_ONLY --> CODE_ONLY_BOX

  CODE_ONLY_BOX --> CODE_ONLY_RESULT{"All passed?"}
  CODE_ONLY_RESULT -- "YES" --> CODE_PASS(["Push proceeds"])
  CODE_ONLY_RESULT -- "NO" --> CODE_ONLY_FAIL(["EXIT 1"])

4. CI Pipeline

Runs on PRs to main or develop. Concurrency group cancels previous runs. Each job has explicit skip conditions. 

flowchart TD
  START(["PR to main or develop
Concurrency: cancel previous"]) --> CHANGES["Job 1: CHANGES
5 min
dorny/paths-filter"]

  CHANGES --> DOCS{"docs-only?
code == false"}
  DOCS -- "YES" --> SKIP_ALL(["All jobs below: SKIPPED
if: docs-only == false"])

  DOCS -- "NO" --> QUALITY["Job 2: QUALITY
10 min
if: docs-only == false"]

  subgraph QUALITY_BOX["Quality Checks (sequential steps)"]
    direction TB
    QA["pnpm format:check"]
    QB["pnpm lint"]
    QC["pnpm typecheck"]
    QD["pnpm sherif
pnpm syncpack:list"]
    QA --> QB --> QC --> QD
  end
  QUALITY --> QUALITY_BOX

  QUALITY_BOX --> UT["Job 3: UNIT TESTS
15 min
needs: [changes, quality]
if: docs-only == false"]
  QUALITY_BOX --> BUILD["Job 4: BUILD
15 min
needs: [changes, quality]
if: docs-only == false"]
  QUALITY_BOX --> BUNDLE["Job 6: BUNDLE ANALYSIS
10 min
needs: [changes, quality]
if: docs-only == false
AND event == pull_request"]

  UT --> UT_ART["Upload: coverage
web + admin + landing
if: always()
7-day retention"]

  BUILD --> BUILD_ART["Upload: .next/
web + admin + landing
if-no-files-found: error
1-day retention"]

  BUILD --> E2E["Job 5: E2E TESTS
30 min
needs: build"]

  E2E --> PW_CACHE{"Playwright
cache hit?"}
  PW_CACHE -- "HIT" --> PW_DEPS["Install deps only
playwright install-deps
all 3 apps"]
  PW_CACHE -- "MISS" --> PW_FULL["Full install
playwright install --with-deps
all 3 apps"]
  PW_DEPS --> E2E_RUN["pnpm test:e2e"]
  PW_FULL --> E2E_RUN

  E2E_RUN --> E2E_ART_OK["Upload: Playwright reports
web + admin + landing
if: always(), 7-day"]
  E2E_RUN --> E2E_ART_FAIL["Upload: test results
web + admin + landing
if: failure(), 7-day"]

  BUNDLE --> BUNDLE_ART["Upload: bundle report
if: always(), 7-day"]

  CHANGES --> DEPLOY_CHG{"deploy == true?
Dockerfile, docker/,
.dockerignore"}
  DEPLOY_CHG -- "YES" --> DOCKER["Job 7: DOCKER SMOKE
20 min
needs: changes
if: deploy == true"]
  DEPLOY_CHG -- "NO" --> DOCKER_SKIP(["Docker: Skipped"])

  DOCKER --> DOCKER_PW{"Playwright
cache hit?"}
  DOCKER_PW -- "HIT" --> DOCKER_DEPS["install-deps"]
  DOCKER_PW -- "MISS" --> DOCKER_FULL["install --with-deps"]
  DOCKER_DEPS --> DOCKER_RUN["Docker health check
+ smoke tests"]
  DOCKER_FULL --> DOCKER_RUN

Quality checks
